function charManipulation()
{
    var sentenceValue = (<HTMLInputElement>document.getElementById("sentenceInput")).value;
        if(!sentenceValue)
        {
            document.getElementById("s1").innerHTML = "Please enter a sentence!";
            return;
        }
        var upperCase = sentenceValue.toUpperCase();
        var lowerCase = sentenceValue.toLocaleLowerCase();
        var revString = "";
        for(let i = sentenceValue.length-1;i>=0;i--)
        {
            revString = revString + sentenceValue[i];
        }

        document.getElementById("s1").innerHTML = "Original: " + sentenceValue;
        document.getElementById("s2").innerHTML = "Uppercase: "+upperCase;
        document.getElementById("s3").innerHTML = "Lowercase: "+lowerCase;
        document.getElementById("s4").innerHTML = "Reversed: " +revString


}
